Merge data frames
-------------------------

### Different cases for the same variables


    (dfNew <- data.frame(id=13:15,
                         group=c("CG", "WL", "T"),
    					 sex=c("f", "f", "m"),
                         age=c(18, 31, 21),
    					 IQ=c(116, 101, 99),
    					 rating=c(4, 4, 1)))

      id group sex age  IQ rating
    1 13    CG   f  18 116      4
    2 14    WL   f  31 101      4
    3 15     T   m  21  99      1

    dfComb <- rbind(myDf, dfNew)
    dfComb[11:15, ]

       id sex group age  IQ rating

### Different variables from the same cases


    (dfA <- data.frame(id=1:4,
                       initials=c("AB", "CD", "EF", "GH"),
                       IV1=c("-", "-", "+", "+"),
    				   DV1=c(10, 10, 11, 14)))

      id initials IV1 DV1
    1  1       AB   -  10
    2  2       CD   -  10
    3  3       EF   +  11
    4  4       GH   +  14

    (dfB <- data.frame(id=1:4,
                       initials=c("AB", "CD", "EF", "GH"),
                       IV2=c("A", "B", "A", "B"),
    				   DV2=c(91, 89, 92, 79)))

      id initials IV2 DV2
    1  1       AB   A  91
    2  2       CD   B  89
    3  3       EF   A  92
    4  4       GH   B  79



    merge(dfA, dfB)

      id initials IV1 DV1 IV2 DV2
    1  1       AB   -  10   A  91
    2  2       CD   -  10   B  89
    3  3       EF   +  11   A  92
    4  4       GH   +  14   B  79

    merge(dfA, dfB, by.x=c(TRUE, FALSE, FALSE, FALSE),
                    by.y=c(TRUE, FALSE, FALSE, FALSE))

      id initials.x IV1 DV1 initials.y IV2 DV2
    1  1         AB   -  10         AB   A  91
    2  2         CD   -  10         CD   B  89
    3  3         EF   +  11         EF   A  92
    4  4         GH   +  14         GH   B  79
